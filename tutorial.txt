#!/bin/bash
apt-get update &> /var/log/update_script.log
apt-get upgrade -y &> /var/log/update_script.log

# Add this on crontab -e :
# 0 4 * * 1 /path/to/file


crontab -l > cronfile
echo "0 4 * * 0 apt-get update >> /var/log/update_script.log && apt-get upgrade >> /var/log/update_script.log" >> cronfile
crontab cronfile
rm cronfile




#!/bin/bash
m1="$(md5sum /etc/crontab)"
m2="8f111d100ea459f68d333d63a8ef2205"

if [ "$m1" != "$m2" ] ; then
	echo "/etc/crontab has been modified in last 24h" | mail -s "Alert, /etc/crontab eddited" root
	exit 1
fi

# Add this on crontab -e :
# 0 0 * * * /path/to/file


#!/bin/bash
crontab -l > cronfile
echo "0 0 * * * \
	md5sum1=\"\$(md5sum /etc/crontab | awk '{ print \$1 }')\" && \
	md5sum2=\"\$(cat /etc/crontab.log)\" && \
	if [ '\$md5sum1' != '\$md5sum2' ] ; then \
		md5sum /etc/crontab | awk '{ print \$1 }' > /etc/crontab.log && \
		echo '/etc/crontab modifier le \$(date)' | mail -s '/etc/crontab modifier' root ; fi" >> cronfile
crontab cronfile
rm cronfile
